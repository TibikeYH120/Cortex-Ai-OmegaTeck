let users = JSON.parse(localStorage.getItem('cx_users') || '{}');
let currentUser = null;

function switchTab(tab) {
  document.getElementById('login-form').style.display = tab === 'login' ? 'block' : 'none';
  document.getElementById('register-form').style.display = tab === 'register' ? 'block' : 'none';
  document.getElementById('tab-login').classList.toggle('active', tab === 'login');
  document.getElementById('tab-register').classList.toggle('active', tab === 'register');
  clearError();
}

function showError(msg) {
  const el = document.getElementById('auth-error');
  el.textContent = '⚠ ' + msg;
  el.classList.add('show');
}
function clearError() {
  document.getElementById('auth-error').classList.remove('show');
}

function togglePass(id, el) {
  const inp = document.getElementById(id);
  inp.type = inp.type === 'password' ? 'text' : 'password';
  el.textContent = inp.type === 'password' ? '👁' : '🙈';
}

function doLogin() {
  clearError();
  const email = document.getElementById('login-email').value.trim();
  const pass = document.getElementById('login-pass').value;
  if (!email || !pass) return showError('Töltsd ki az összes mezőt!');
  const user = users[email];
  if (!user || user.pass !== btoa(pass)) return showError('Hibás email vagy jelszó!');
  loginAs(user);
}

function doRegister() {
  clearError();
  const name = document.getElementById('reg-name').value.trim();
  const email = document.getElementById('reg-email').value.trim();
  const pass = document.getElementById('reg-pass').value;
  if (!name || !email || !pass) return showError('Töltsd ki az összes mezőt!');
  if (pass.length < 6) return showError('A jelszó legalább 6 karakter legyen!');
  if (!/\S+@\S+\.\S+/.test(email)) return showError('Érvénytelen email cím!');
  if (users[email]) return showError('Ez az email már foglalt!');
  users[email] = { name, email, pass: btoa(pass) };
  localStorage.setItem('cx_users', JSON.stringify(users));
  loginAs(users[email]);
}

function loginGuest() {
  loginAs({ name: 'Vendég', email: 'guest@cortex.ai' });
}

function loginAs(user) {
  currentUser = user;
  document.getElementById('auth-screen').style.animation = 'authFadeOut 0.4s ease both';
  setTimeout(() => {
    document.getElementById('auth-screen').style.display = 'none';
    const app = document.getElementById('app');
    app.classList.add('visible');
    initApp();
  }, 380);
}

function doLogout() {
  currentUser = null;
  document.getElementById('app').classList.remove('visible');
  document.getElementById('app').style.display = 'none';
  const auth = document.getElementById('auth-screen');
  auth.style.display = 'flex';
  auth.style.animation = 'authFadeIn 0.4s ease both';
  document.getElementById('login-email').value = '';
  document.getElementById('login-pass').value = '';
  closeProfileMenu();
}

function createSession(firstMsg) {
  const id = Date.now().toString();

  renderMessages();
  closeMobileSidebar();
}

function renderMessages() {
  if (!activeSession) return showWelcome();
  const inner = document.getElementById('msgs-inner');
  inner.innerHTML = activeSession.messages.map(m => buildBubble(m.role, m.content)).join('');

function buildBubble(role, content) {
  const isAI = role === 'assistant';
  const avatarLabel = isAI ? 'CX' : (currentUser?.name?.[0]?.toUpperCase() || 'U');
  const avatarClass = isAI ? 'ai' : 'human';
  const nameLabel = isAI ? 'Cortex AI' : (currentUser?.name || 'Te');
  const nameClass = isAI ? 'ai-name' : 'user-name';
  const now = new Date().toLocaleTimeString('hu',{hour:'2-digit',minute:'2-digit'});
  return `
    <div class="msg ${role}">
      <div class="msg-avatar ${avatarClass}">${avatarLabel}</div>
      <div class="msg-body">
        <div class="msg-meta">
          <span class="msg-name ${nameClass}">${nameLabel}</span>
          <span class="msg-time">${now}</span>
        </div>
        <div class="bubble">${formatText(content)}</div>
      </div>
    </div>`;
}

function formatText(txt) {
  txt = txt.replace(/```(\w+)?\n?([\s\S]*?)```/g, (_,lang,code) => `<pre><code>${escHtml(code.trim())}</code></pre>`);

txt = txt.replace(/`([^`]+)`/g, '<code>$1</code>'); txt = txt.replace(/\*\*(.*?)\*\*/g, '<strong>$1</strong>'); txt = txt.replace(/^### (.*)/gm, '<h3>$1</h3>'); txt = txt.replace(/^- (.*)/gm, '<li>$1</li>'); txt = txt.replace(/<li>([\s\S]*?)<\/li>/gs, m => `<ul>${m}</ul>`); const parts = txt.split(/\n\n+/); return parts.length > 1 ? parts.map(p=>`<p>${p.replace(/\n/g,’<br>’)}</p>`).join('') : `<p>${txt.replace(/\n/g,’<br>’)}</p>`;
}

function escHtml(t) { return t.replace(/&/g,’&’).replace(/</g,’<’).replace(/>/g,’>’); }

function scrollBottom() {
const el = document.getElementById(‘messages’);
setTimeout(() => el.scrollTop = el.scrollHeight, 50);
}

async function sendMsg() {
if (isLoading) return;
const inp = document.getElementById(‘chat-input’);
const text = inp.value.trim();
if (!text) return;

activeSession = createSession(text);
document.getElementById(‘chat-title’).innerHTML = escHtml(activeSession.title.slice(0,30));
}

activeSession.messages.push({ role:‘user’, content: text });
saveSessions();
inp.value = ‘’; inp.style.height = ‘auto’;
isLoading = true; document.getElementById(‘send-btn’).disabled = true;

// Render user msg
const inner = document.getElementById(‘msgs-inner’);
// Remove welcome if present
const welcome = inner.querySelector(’.welcome-screen’);
if (welcome) welcome.remove();

inner.insertAdjacentHTML(‘beforeend’, buildBubble(‘user’, text));
scrollBottom();
